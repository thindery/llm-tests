Video Assembly Pipeline for Sleep Stories

import ffmpeg
import json
from pathlib import Path

class SleepStoryAssembler:
    def __init__(self, project_dir):
        self.project = Path(project_dir)
        self.chapters = self.load_chapters()
        
    def load_chapters(self):
        with open(self.project / "script.json") as f:
            return json.load(f)["chapters"]
    
    def calculate_image_duration(self, chapter_duration, num_images):
        """Each image shows for chapter_duration / num_images"""
        return chapter_duration / num_images
    
    def create_ken_burns_clip(self, image_path, duration, output_path):
        """Apply slow zoom + pan to static image"""
        # Randomize zoom direction (1.0 â†’ 1.1 or 1.1 â†’ 1.0)
        zoom_start = 1.0
        zoom_end = 1.08
        pan_x = "(iw-iw/zoom)*0.5"
        pan_y = "(ih-ih/zoom)*0.5"
        
        (
            ffmpeg
            .input(image_path, loop=1, t=duration)
            .filter('zoompan', 
                    z=f"min(zoom+0.0015,{zoom_end})", 
                    x=pan_x, y=pan_y,
                    d=duration*24,  # frames at 24fps
                    s="1920x1080")
            .output(output_path, vcodec='libx264', preset='medium', crf=23)
            .run()
        )
    
    def crossfade_chapter_clips(self, clip_paths, output_path):
        """Join clips with 30-second crossfade between each"""
        crossfade_duration = 30  # seconds
        
        # Build complex filter for crossfades
        stream = ffmpeg.input(clip_paths[0]).video
        audio_stream = ffmpeg.input(clip_paths[0]).audio
        
        for i, clip in enumerate(clip_paths[1:], 1):
            next_video = ffmpeg.input(clip).video
            next_audio = ffmpeg.input(clip).audio
            
            # Concatenate with overlap
            stream = ffmpeg.filter(
                [stream, next_video],
                'xfade',
                transition='fade',
                duration=crossfade_duration,
                offset=f"duration-{crossfade_duration}"
            )
            
            # Simple audio crossfade
            audio_stream = ffmpeg.filter(
                [audio_stream, next_audio],
                'acrossfade',
                durations=crossfade_duration
            )
        
        # Output chapter video
        ffmpeg.output(stream, audio_stream, output_path).run()
    
    def mix_audio_layers(self, narration_path, ambient_paths, output_path):
        """Mix narration + multiple ambient tracks"""
        narration = ffmpeg.input(narration_path).audio
        
        # Mix all inputs
        mixed = narration
        for ambient in ambient_paths:
            ambient_input = ffmpeg.input(ambient).audio
            # Ambient at 30% volume
            ambient_input = ffmpeg.filter(ambient_input, 'volume', 0.3)
            mixed = ffmpeg.filter([mixed, ambient_input], 'amix', inputs=2)
        
        ffmpeg.output(mixed, output_path, ac=2, ar=48000).run()

        pass
    
    def assemble_full_video(self):
        """Master assembly pipeline"""
        chapter_videos = []
        
        for chapter in self.chapters:
            # Generate image clips with Ken Burns
            image_clips = []
            image_duration = self.calculate_image_duration(
                chapter["duration"], 
                len(chapter["images"])
            )
            
            for img in chapter["images"]:
                clip_path = self.project / "temp" / f"{img.stem}.mp4"
                self.create_ken_burns_clip(img, image_duration, clip_path)
                image_clips.append(clip_path)

# quality_gates.py

class QualityGate:
    """Pre-upload verification system"""
    
    GATES = [
        {
            "name": "audio_duration_check",
            "check": lambda video: abs(video.audio_duration - video.target_duration) < 60,
            "fail_action": "retry_generation"
        },
        {
            "name": "audio_quality_check", 
            "check": lambda video: video.audio_lufs > -20 and video.audio_lufs < -14,
            "fail_action": "re_normalize"
        },
        {
            "name": "video_integrity_check",
            "check": lambda video: video.frame_count == video.expected_frames,
            "fail_action": "re_render"
        },
        {
            "name": "thumbnail_exists",
            "check": lambda video: video.thumbnail_file.exists() and video.thumbnail_size < 2_000_000,
            "fail_action": "generate_thumbnail"
        },
        {
            "name": "silent_segment_check",
            "check": lambda video: video.detect_silent_segments() < 3,
            "fail_action": "flag_for_review"
        }
    ]
    
    def evaluate(self, video):
        results = []
        for gate in self.GATES:
            passed = gate["check"](video)
            results.append({
                "gate": gate["name"],
                "passed": passed,
                "action": None if passed else gate["fail_action"]
            })
        
        all_passed = all(r["passed"] for r in results)
        return {
            "approved": all_passed,
            "results": results,
            "retry_actions": [r["action"] for r in results if not r["passed"]]
        }
```

    zoompan=z='min(zoom+0.08,1.2)':x='iw/2-(iw/zoom/2)':y='ih/2-(ih/zoom/2)':d=360:s=1920x1080:fps=24,
    fade=t=in:st=0:d=0.5,

    [vo][ti][ra][pt][rt]amix=inputs=5:duration=first[aout]
  " -map "[aout]" -ac 2 -ar 48000 audio_mixed.mp3

# 4. Final mux
ffmpeg -i visuals.mp4 -i audio_mixed.mp3 -c:v copy -c:a aac -b:a 192k -shortest sample_output.mp4
```
